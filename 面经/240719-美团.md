   ```vue
   1.实现一个倒计时组件，并且自动开始记时，到0停止计时
   <template>
     <div>
       <p>{{ timeLeft }}</p>
     </div>
   </template>
   
   <script>
   export default {
     name: 'CountdownTimer',
     props: {
       startTime: {
         type: Number,
         required: true
       }
     },
     data() {
       return {
         timeLeft: this.startTime
       };
     },
     mounted() {
       this.startCountdown();
     },
     methods: {
       startCountdown() {
         const countdownInterval = setInterval(() => {
           if (this.timeLeft > 0) {
             this.timeLeft--;
           } else {
             clearInterval(countdownInterval);
           }
         }, 1000);
       }
     }
   };
   </script>
   ```
