查找某个节点的路径的方法通常有两种，一种是递归算法,另一种是非递归算法

``` 
// 定义树节点
class TreeNode {
	constructor(value) {
		this.root = value;
		this.left = null;
		this.left = null;
	}
}
// 构建树
let root = new TreeNode(1);
root.left = new TreeNode(2);
root.right = new TreeNode(3);
root.left.left = new TreeNode(4);
root.right.right = new TreeNode(5);

// 递归算法
// 递归中序遍历二叉树
function midOrder(root){
	if(!root) return;
	// 递归访问左子树
	midOrder(root.left);
	// 递归访问右子树
	midOrder(root.right);
}
// 非递归中序遍历二叉树
function midOrderN(root){
	let p = root; // 当前遍历的结点，初始为根节点
	let arr = [];
	while(p || arr.length!==0){
		if(p){
			// 遍历左子树
			arr.push(p);
			p = p.left;
		} else{
			// p为空，出栈
			let node = arr.pop();
			// 访问该节点
			console.log(node.value);
			p = node.right;
		}
	}
}
```
